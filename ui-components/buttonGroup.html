CSS 类用法 ​

使用预定义的CSS类来构建按钮组

查看代码
html
<!-- 基础按钮组 -->
<div class="button-group">
  <button class="button">
    左按钮
  </button>
  <button class="button">
    中按钮
  </button>
  <button class="button">
    右按钮
  </button>
</div>

<!-- 主要按钮组 -->
<div class="button-group">
  <button class="button button-primary">
    左按钮
  </button>
  <button class="button button-primary">
    中按钮
  </button>
  <button class="button button-primary">
    右按钮
  </button>
</div>
css
.button-group {
  @apply inline-flex;

  .button {
    &:first-child {
      @apply rounded-r-none;
    }

    &:last-child {
      @apply rounded-l-none;
    }

    &:not(:first-child) {
      @apply ml-[-1px];
    }

    &:not(:first-child):not(:last-child) {
      @apply rounded-none;
    }

    &:not(:disabled):hover {
      z-index: 2;
    }
  }
}
